% 1. The Image Macro: added trim and M-alignment
\newcommand{\Img}[1]{%
    \includegraphics[
        trim=0 0 0 0, clip, % Change to trim=10 10 10 10 if you want to force-crop edges
        valign=M,
        max height=2cm,
        keepaspectratio
    ]{\detokenize{/app/media/items/#1}}%
}

            {\tiny \MakeUppercase{#2}} \par \vspace{1mm}
            {\large \textbf{#3}} \par \vspace{1mm}
            {\small #4} \par \vspace{4mm}
            {\Large \textbf{#5~kr.}} \vspace{2mm}
        \end{tabularx}
    \end{minipage}%
}\hspace{0.4cm}}
